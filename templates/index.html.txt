    <h2>✈️ Путешествия исследователей мира</h2>
    <div id="tripsContainer"></div>

    <script>
        // Проверка авторизации из шаблона Jinja2
        const loggedUser = "{{ user.username if user else '' }}";

        if (loggedUser) {
            showUserInterface(loggedUser);
        }

        function showUserInterface(username) {
            document.getElementById('authSection').style.display = 'none';
            document.getElementById('userGreeting').style.display = 'block';
            document.getElementById('currentUserName').innerText = username;
            document.getElementById('createTripSection').style.display = 'block';
        }

        async function login() {
            const username = document.getElementById('usernameInput').value.trim();
            if(!username) return alert("Введите имя!");
            const formData = new FormData();
            formData.append('username', username);
            await fetch('/auth/login', { method: 'POST', body: formData });
            location.reload();
        }

        async function logout() {
            await fetch('/auth/logout', { method: 'POST' });
            location.reload();
        }

        async function submitTrip() {
            const formData = new FormData(document.getElementById('tripForm'));
            await fetch('/trips/create', { method: 'POST', body: formData });
            document.getElementById('tripForm').reset();
            loadTrips();
        }

        async function loadTrips() {
            const res = await fetch('/trips');
            const trips = await res.get_json() || await res.json();
            const container = document.getElementById('tripsContainer');
            container.innerHTML = '';
            
            trips.forEach(t => {
                container.innerHTML += `
                    <div class="block trip-card">
                        <h3>${t.title}</h3>
                        <p>${t.description}</p>
                        <div style="margin: 10px 0;">
                            <span class="badge">💰 Бюджет: ${t.cost.toLocaleString()} руб.</span>
                            <span class="badge" style="background:#e8f5e9; color:#2e7d32;">📍 Локации: ${t.places_to_visit || 'Не указаны'}</span>
                        </div>
                        <div style="font-size: 13px; margin-top:5px;">
                            🛡️ Безопасность: <span class="rating">${t.safety_rating}/5</span> | 
                            🚗 Транспорт: <span class="rating">${t.transport_rating}/5</span>
                        </div>
                        <p style="text-align:right; margin:0;"><small>Автор дневника: <b>@${t.author}</b></small></p>
                    </div>
                `;
            });
        }

        loadTrips();
    </script>
