from fastapi import FastAPI, Depends, HTTPException, Form, Request, Response
from fastapi.responses import HTMLResponse
from fastapi.templating import Jinja2Templates
from sqlalchemy.orm import Session
import database as db

app = FastAPI()
db.init_db()

templates = Jinja2Templates(directory="templates")

def get_db():
    session = db.SessionLocal()
    try: yield session
    finally: session.close()

# Определение текущего пользователя системы через Cookie сессию
def get_current_user(request: Request, session: Session = Depends(get_db)):
    username = request.cookies.get("travel_user")
    if not username: return None
    return session.query(db.User).filter(db.User.username == username).first()

@app.get("/", response_class=HTMLResponse)
async def index(request: Request, session: Session = Depends(get_db)):
    user = get_current_user(request, session)
    return templates.TemplateResponse("index.html", {"request": request, "user": user})

# ➔ Создать пользователя системы (Вход / Регистрация)
@app.post("/auth/login")
async def login(response: Response, username: str = Form(...), session: Session = Depends(get_db)):
    user = session.query(db.User).filter(db.User.username == username).first()
    if not user:
        user = db.User(username=username)
        session.add(user)
        session.commit()
    response.set_cookie(key="travel_user", value=username)
    return {"status": "success", "username": username}

@app.post("/auth/logout")
async def logout(response: Response):
    response.delete_cookie("travel_user")
    return {"status": "logged_out"}

# ➔ Создать функцию просмотра путешествий других пользователей (Все записи)
@app.get("/trips")
async def get_all_trips(session: Session = Depends(get_db)):
    trips = session.query(db.Trip).all()
    return [{
        "id": t.id, "title": t.title, "description": t.description, "cost": t.cost,
        "places_to_visit": t.places_to_visit, "safety_rating": t.safety_rating,
        "transport_rating": t.transport_rating, "author": t.author.username
    } for t in trips]

# ➔ Создать для пользователя функцию записи своего путешествия
@app.post("/trips/create")
async def create_trip(
    title: str = Form(...), description: str = Form(...), cost: float = Form(0.0),
    places: str = Form(""), safety: int = Form(5), transport: int = Form(5),
    user: db.User = Depends(get_current_user), session: Session = Depends(get_db)
):
    if not user: raise HTTPException(status_code=401, detail="Необходима авторизация")
    
    new_trip = db.Trip(
        title=title, description=description, cost=cost,
        places_to_visit=places, safety_rating=safety, transport_rating=transport,
        author_id=user.id
    )
    session.add(new_trip)
    session.commit()
    return {"status": "success"}
